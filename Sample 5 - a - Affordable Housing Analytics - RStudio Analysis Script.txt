## This script regresses percentile ranks of delinquency status against percentile 
## ranks of borrowers’ FICO scores, borrowers’ combined loan-to-value ratios, 
## borrowers’ debt-to-income ratios, and properties’ coastal state status, for the 
## mortgages in Freddie Mac’s Home Possible program and mortgages that fit a 
## similar risk profile outside the Home Possible Program. Mortgage data has been 
## provided by Milliman in an Apache Parquet file called all_data.parquet. 
## Properties are assigned a coastal state status flag of 1 if they are in states 
## on the Gulf of Mexico or Atlantic Ocean coastline, or in HI, WV, DC, PA, or VT; 
## they are assigned 0 otherwise. Delinquency statuses are numeric and have 
## historically ranged from 0 to 6, where 0 denotes non-delinquency. For FICO 
## scores, the percentile ranks converge to 100% for high scores and 0% for low 
## scores; for CLTV ratios, DTI ratios, coastal state statuses, and delinquency 
## statuses, the percentile ranks converge to 100% for low scores and 0% for high 
## scores. Replicated values receive the average of the percentiles the values 
## would have been assigned had they been distinct – for example, if 60% of the 
## FICO scores in the data were 680 and 40% of the FICO scores were 770, then all 
## the 680 FICO scores would be assigned the 30th percentile (the average or 
## midpoint of the 0 and 60th percentiles), and all the 770 FICO scores would be
## assigned the 80th percentile (the average of the 60th and 100th percentiles).
##
## The regression is a least-squares fit that determines coefficients and an
## intercept for y = m[1]x[1] + m[2]x[2] + m[3]x[3] + m[4]x[4] + b:
## 
## y: delinquency status percentile
## x[1]: FICO score percentile
## x[2]: DTI ratio percentile
## x[3]: CLTV ratio percentile
## x[4]: Coastal state status percentile
## 
## Protecdiv’s mortgage risk scoring is based on the weights m[1], m[2], m[3], 
## and m[4] that result from the least-squares regression:
## 
## w[n] = m[n] / (m[1]+m[2]+m[3]+m[4]), n ∈ {1, 2, 3, 4}

## Import Milliman's mortgage data and the BEA's metropolitan statistical
## area-level income data

all_data <- read_parquet("/Users/cateydel/Documents/Protecdiv/Clients and Prospects/Freddie Mac/From Milliman/all_data.parquet", as_tibble = TRUE)
CAINC1_MSA_1969_2019 <- read.csv("/Users/cateydel/Documents/Protecdiv/Clients and Prospects/Freddie Mac/Bureau of Economic Affairs/CAINC1/CAINC1_MSA_1969_2019.csv")


## Isolate CY 2019 per capita income in the BEA data

msaPerCapitaIncome <- CAINC1_MSA_1969_2019 %>% filter(grepl('Dollars', Unit))
msaPerCapitaIncome_2019 <- msaPerCapitaIncome %>% mutate(GeoFIPSChar=gsub(" ","",as.character(GeoFIPS))) %>% mutate(GeoFIPSChar4=substr(GeoFIPSChar,1,4)) %>% select(c(GeoFIPS, GeoFIPSChar, GeoFIPSChar4, X2019))


## Build a table of the most current records (i.e., the records with the most
## mature loan ages) for each mortgage. Milliman's file is a compilation of
## Freddie Mac's mortgage performance files, which show each mortgage's monthly
## development. We don't need to see all the development, just the latest.

all_data_selCols <- all_data %>% mutate(msaLength = nchar(MSA)) %>% select(c(LoanID, State, LoanAge, MSA, CLTV, FICO, DTI, UPB, DelqStatus, delq_status, minority_perc, msaLength)) %>% mutate(MSA4=substr(MSA,1,4))
all_data_selCols_maxAge <- all_data_selCols %>% group_by(LoanID) %>% summarize(maxAge = max(LoanAge, na.rm=TRUE))
all_data_selCols_latest <- inner_join(all_data_selCols, all_data_selCols_maxAge, by = c("LoanID" = "LoanID", "LoanAge" = "maxAge"))


## Join the latest-records table with the MSA-level per capita income table

all_data_selCols_latest_wInc <- left_join(all_data_selCols_latest, msaPerCapitaIncome_2019, by = c("MSA4" = "GeoFIPSChar4")) %>% mutate(mmUpbTimesAvgInc = 0.000001*UPB*X2019) %>% mutate(upb_nonnullAvgInc=UPB*as.integer(!is.na(X2019))) %>% mutate(mmUpbTimesMinPct = 0.000001*UPB*minority_perc) %>% mutate(upb_nonnullMinPct=UPB*as.integer(!is.na(minority_perc)))


## Calculate UPB-weighted averages of minority population percentages and per
## capita income for the full portfolio

all_data_selCols_avgInc <- all_data_selCols_latest_wInc %>% summarize( sumUpb = sum(upb_nonnullAvgInc, na.rm=TRUE), sumMmUpbTimesInc = sum(mmUpbTimesAvgInc, na.rm=TRUE), sumUpb_MinPct = sum(upb_nonnullMinPct, na.rm=TRUE), sumMmUpbTimesMinPct = sum(mmUpbTimesMinPct, na.rm=TRUE)) %>% mutate(avgInc = (sumMmUpbTimesInc/sumUpb)*1000000) %>% mutate(avgMinPct = (sumMmUpbTimesMinPct/sumUpb_MinPct)*1000000)


## For mortgages in MSAs that don't have a reported MSA-level per capita income,
## use the portfolio average as a placeholder

all_data_selCols_upd0 <- inner_join(filter(all_data_selCols_latest_wInc, is.na(X2019)), all_data_selCols_avgInc, by=character()) %>% select(c(LoanID, State, LoanAge, MSA, CLTV, FICO, DTI, UPB, DelqStatus, delq_status, minority_perc, avgInc))
all_data_selCols_upd1 <- inner_join(filter(all_data_selCols_latest_wInc, !is.na(X2019)), all_data_selCols_avgInc, by=character()) %>% select(c(LoanID, State, LoanAge, MSA, CLTV, FICO, DTI, UPB, DelqStatus, delq_status, minority_perc, X2019))
all_data_selCols_upd1 <- rename(all_data_selCols_upd1, avgInc=X2019)
all_data_selCols_latest_wInc <- bind_rows(all_data_selCols_upd1, all_data_selCols_upd0)


## For mortgages in MSAs that don't have a reported minority population percentage,
## use the portfolio average as a placeholder

all_data_selCols_upd0 <- inner_join(filter(all_data_selCols_latest_wInc, is.na(minority_perc)), rename(all_data_selCols_avgInc, avg_inc=avgInc), by=character()) %>% select(c(LoanID, State, LoanAge, MSA, CLTV, FICO, DTI, UPB, DelqStatus, delq_status, avgInc, avgMinPct))
all_data_selCols_upd1 <- inner_join(filter(all_data_selCols_latest_wInc, !is.na(minority_perc)), rename(all_data_selCols_avgInc, avg_inc=avgInc), by=character()) %>% select(c(LoanID, State, LoanAge, MSA, CLTV, FICO, DTI, UPB, DelqStatus, delq_status, avgInc, minority_perc))
all_data_selCols_upd0 <- rename(all_data_selCols_upd0, minority_perc=avgMinPct)
all_data_selCols_latest_wInc <- bind_rows(all_data_selCols_upd1, all_data_selCols_upd0)


## Define the coastal state status flag

all_data_selCols_latest_wInc <- all_data_selCols_latest_wInc %>%
  mutate(coastalState = case_when(
    State=="HI" | State=="TX" | State=="LA" | State=="MS" | State=="AL" | State=="FL" | State=="GA" | State=="SC" | State=="NC" | State=="VA" | State=="WV" | State=="DC" | State=="MD" | State=="DE" | State=="NJ" | State=="PA" | State=="NY" | State=="CT" | State=="RI" | State=="MA" | State=="VT" | State=="NH" | State=="ME" ~ 1,
    State!="HI" & State!="TX" & State!="LA" & State!="MS" & State!="AL" & State!="FL" & State!="GA" & State!="SC" & State!="NC" & State!="VA" & State!="WV" & State!="DC" & State!="MD" & State!="DE" & State!="NJ" & State!="PA" & State!="NY" & State!="CT" & State!="RI" & State!="MA" & State!="VT" & State!="NH" & State!="ME" ~ 0
  ))


## Compute the FICO, coastal state status, CLTV, DTI, and delinquency
## status percentiles

rankWork_DTI <- all_data_selCols_latest_wInc %>% select(c(LoanID, DTI))
rankWork_pctiles_DTI <- inner_join(rankWork_DTI %>% mutate(rk_DTI=rank(DTI)), rankWork_DTI %>% group_by(DTI) %>% summarize(ct_DTI=n()) %>% mutate(nr=nrow(rankWork_DTI)), by="DTI") %>% mutate(pctRk_DTI=1-rk_DTI/(1+nr))

rankWork_FICO <- all_data_selCols_latest_wInc %>% select(c(LoanID, FICO))
rankWork_pctiles_FICO <- inner_join(rankWork_FICO %>% mutate(rk_FICO=rank(FICO)), rankWork_FICO %>% group_by(FICO) %>% summarize(ct_FICO=n()) %>% mutate(nr=nrow(rankWork_FICO)), by="FICO") %>% mutate(pctRk_FICO=rk_FICO/(1+nr))

rankWork_CLTV <- all_data_selCols_latest_wInc %>% select(c(LoanID, CLTV))
rankWork_pctiles_CLTV <- inner_join(rankWork_CLTV %>% mutate(rk_CLTV=rank(CLTV)), rankWork_CLTV %>% group_by(CLTV) %>% summarize(ct_CLTV=n()) %>% mutate(nr=nrow(rankWork_CLTV)), by="CLTV") %>% mutate(pctRk_CLTV=1-rk_CLTV/(1+nr))

rankWork_coastalState <- all_data_selCols_latest_wInc %>% select(c(LoanID, coastalState))
rankWork_pctiles_coastalState <- inner_join(rankWork_coastalState %>% mutate(rk_coastalState=rank(coastalState)), rankWork_coastalState %>% group_by(coastalState) %>% summarize(ct_coastalState=n()) %>% mutate(nr=nrow(rankWork_coastalState)), by="coastalState") %>% mutate(pctRk_coastalState=1-rk_coastalState/(1+nr))

rankWork_delq_status <- all_data_selCols_latest_wInc %>% select(c(LoanID, delq_status))
rankWork_pctiles_delq_status <- inner_join(rankWork_delq_status %>% mutate(rk_delq_status=rank(delq_status)), rankWork_delq_status %>% group_by(delq_status) %>% summarize(ct_delq_status=n()) %>% mutate(nr=nrow(rankWork_delq_status)), by="delq_status") %>% mutate(pctRk_delq_status=1-rk_delq_status/(1+nr))

all_data_selCols_pctiles <- inner_join(inner_join(inner_join(rankWork_pctiles_delq_status, rankWork_pctiles_FICO %>% select(!c(nr)), by="LoanID"), rankWork_pctiles_DTI %>% select(!c(nr)), by="LoanID"), inner_join(rankWork_pctiles_CLTV, rankWork_pctiles_coastalState %>% select(!c(nr)), by="LoanID"))


## Regress the FICO, coastal state status, CLTV, and DTI percentiles against the
## delinquency status percentiles

linear_model_test <- lm(pctRk_delq_status ~ pctRk_FICO + pctRk_coastalState + pctRk_CLTV + pctRk_DTI, data=all_data_selCols_pctiles)
summary(linear_model_test)


## Check 2,000 random records

hp_and_comp_percentiles_2000records <- sample_n(all_data_selCols_pctiles, 2000)
write.csv(hp_and_comp_percentiles_2000records, file = "/Users/cateydel/Documents/Protecdiv/Clients and Prospects/Freddie Mac/Milliman Data Exploration/hp_and_comp_percentiles_2000records.csv")


## Build a table of risk score weights

tidy_linear_model_test <- tidy(linear_model_test)
tidy_linear_model_test_weights <- filter(tidy_linear_model_test, term!="(Intercept)") %>% mutate(coeffWt=round(100*estimate/sum(estimate),1))


## Calculate risk scores for each mortgages

all_data_riskScoresByLoanID <- inner_join(inner_join(inner_join(all_data_selCols_pctiles, filter(tidy_linear_model_test_weights, term=='pctRk_FICO'), by=character()) %>% mutate(RS1=pctRk_FICO*coeffWt) %>% select(c(LoanID, RS1)), inner_join(all_data_selCols_pctiles, filter(tidy_linear_model_test_weights, term=='pctRk_DTI'), by=character()) %>% mutate(RS2=pctRk_DTI*coeffWt) %>% select(c(LoanID, RS2)), by="LoanID"), inner_join(inner_join(all_data_selCols_pctiles, filter(tidy_linear_model_test_weights, term=='pctRk_CLTV'), by=character()) %>% mutate(RS3=pctRk_CLTV*coeffWt) %>% select(c(LoanID, RS3)), inner_join(all_data_selCols_pctiles, filter(tidy_linear_model_test_weights, term=='pctRk_coastalState'), by=character()) %>% mutate(RS4=pctRk_coastalState*coeffWt) %>% select(c(LoanID, RS4)), by="LoanID"), by="LoanID") %>% mutate(riskScore=round(RS1+RS2+RS3+RS4,1))


## Build a dataframe of mortgages with 40+ risk scores, and either <$48,000 average
## MSA-level incomes or 63%+ minority populations  

data_highRiskScore_and_lowIncOrHighMino <- inner_join(filter(all_data_riskScoresByLoanID, riskScore>=40), filter(all_data_selCols_latest_wInc, avgInc<48000 | minority_perc>=0.63), by="LoanID")


## Calculate UPB-weighted delinquency probabilities by risk score

data_delinquenciesByRiskScore <- data_highRiskScore_and_lowIncOrHighMino %>% mutate(upbTimesDelqSign=UPB*sign(delq_status)) %>% group_by(riskScore) %>% summarize(UPB = sum(UPB), upbTimesDelqSign=sum(upbTimesDelqSign), delqRatio=sum(upbTimesDelqSign)/sum(UPB))
write.csv(data_delinquenciesByRiskScore, file = "/Users/cateydel/Documents/Protecdiv/Clients and Prospects/Freddie Mac/Milliman Data Exploration/data_delinquenciesByRiskScore.csv")
